// Client-side lazy loading
function loadTabContent(tabId) {
    if (!tabCache[tabId]) {
        showLoadingSpinner(tabId);
        
        $.ajax({
            url: '/Controls/Dashboard/' + tabId + 'Dashboard.aspx',
            type: 'GET',
            cache: true,
            success: function(data) {
                tabCache[tabId] = data;
                $('#' + tabId + '-content').html(data);
                hideLoadingSpinner(tabId);
                initializeWidgets(tabId);
            }
        });
    } else {
        // Use cached content
        $('#' + tabId + '-content').html(tabCache[tabId]);
    }
}
